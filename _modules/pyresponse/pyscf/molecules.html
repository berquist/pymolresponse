[docs]def molecule_water_sto3g(verbose: int = 0) -> pyscf.gto.Mole:

    mol = pyscf.gto.Mole()
    mol.verbose = verbose
    mol.output = None

    with open(COORDDIR / "water.xyz") as fh:
        mol.atom = fh.read()
    mol.basis = "sto-3g"
    mol.charge = 0
    mol.spin = 0

    mol.unit = "Bohr"

    return mol
